\makeatletter
    \def\@maketitle{%
  \newpage
  \null
  % \vskip 2em%
  \vspace{-8ex}
  \begin{center}%
  \let \footnote \thanks
    \vskip 2pt
    \bfseries{\textsc{\large\@title - \@author}}\par
    \vspace{-2ex}\@date
  \end{center}%
  \par
  % \vskip 1.5em
  }
\makeatother
