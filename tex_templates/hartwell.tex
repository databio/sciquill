% Redefine maketitle
\makeatletter
\def\@maketitle{%
  \newpage
  \null
  \vskip 2em%
  \begin{center}%
  \let \footnote \thanks
    {\huge \@title \par}%
    \vskip 3in%
    {
      \lineskip .5em%
        \Large $author$  \\
        \large $jobtitle$  \\
        $for(depts)$
        $depts$  \\
        $endfor$
        \vskip 2em%
        \normalsize
        $for(addresslines)$
        $addresslines$  \\
        $endfor$        
      }%
    \vskip 4em%
    {\large \@date}
  \end{center}%
  \par
  \vskip 1.5em}
\makeatother
